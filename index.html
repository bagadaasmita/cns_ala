#from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import rsa, padding
from cryptography.hazmat.primitives import serialization

# Step 1: Generate Keys
private_key = rsa.generate_private_key(
    public_exponent=65537,
    key_size=2048
)

public_key = private_key.public_key()

# Step 2: Message
message = b"Hello, this is a digital signature example"

# Step 3: Sign the message
signature = private_key.sign(
    message,
    padding.PSS(
        mgf=padding.MGF1(hashes.SHA256()),
        salt_length=padding.PSS.MAX_LENGTH
    ),
    hashes.SHA256()
)

print("Signature created successfully!")

# Step 4: Verify Signature
try:
    public_key.verify(
        signature,
        message,
        padding.PSS(
            mgf=padding.MGF1(hashes.SHA256()),
            salt_length=padding.PSS.MAX_LENGTH
        ),
        hashes.SHA256()
    )
    print("Signature is VALID ✅")
except:
    print("Signature is INVALID ❌")

# import hashlib

def generate_hashes(message):
    print("\nOriginal Message:", message)

    sha1 = hashlib.sha1(message.encode()).hexdigest()
    sha256 = hashlib.sha256(message.encode()).hexdigest()
    sha512 = hashlib.sha512(message.encode()).hexdigest()

    print("\nSHA-1   :", sha1)
    print("SHA-256 :", sha256)
    print("SHA-512 :", sha512)

# Original message
msg1 = "Hello World"

# Slightly modified message (small change)
msg2 = "Hello world"   # 'W' changed to 'w'

print("----- First Input -----")
generate_hashes(msg1)

print("\n\n----- Modified Input -----")
generate_hashes(msg2)

import hmac
import hashlib

# Shared secret key (same as sender)
secret_key = b'supersecretkey'

# Received message and MAC
received_message = b'Hello, this is a secure message'
received_mac = input("Enter received MAC: ")

# Recompute MAC
new_mac = hmac.new(secret_key, received_message, hashlib.sha256).hexdigest()

print("Recomputed MAC:", new_mac)

# Verification
if hmac.compare_digest(received_mac, new_mac):
    print("Message is AUTHENTIC ✅")
else:
    print("Message is TAMPERED ❌")
